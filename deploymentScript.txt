    <script type="text/javascript">
        //The webpageindex will be specific to each site
        window.webpageIndex = 'WEBSITEID';
        window.usaProxyServerIP = 'CAPTURESERVERADDRESS:2727';
        window.sessionId = getCookie("proxyUserID");

        window.usaProxySrc = "http://" + window.usaProxyServerIP + "/usaproxylolo/file/proxyscript.js?sd=" + window.webpageIndex + "&sid=" + window.sessionId

        //we add the script dinamically
        var usaProxyScriptNode = document.createElement('script');
        usaProxyScriptNode.id = 'proxyScript_UsaProxy';
        usaProxyScriptNode.type = 'text/javascript';
        usaProxyScriptNode.src = window.usaProxySrc;

        document.getElementsByTagName('head')[0].appendChild(usaProxyScriptNode);

        function getCookie(c_name) {
            var i, x, y, ARRcookies = document.cookie.split(";");
            for (i = 0; i < ARRcookies.length; i++) {
                x = ARRcookies[i].substr(0, ARRcookies[i].indexOf("="));
                y = ARRcookies[i].substr(ARRcookies[i].indexOf("=") + 1);
                x = x.replace(/^\s+|\s+$/g, "");
                if (x == c_name) {
                    return unescape(y);
                }
            }

            //we didn't find the cookie, so we return null
            return "null";
        }
    </script>
